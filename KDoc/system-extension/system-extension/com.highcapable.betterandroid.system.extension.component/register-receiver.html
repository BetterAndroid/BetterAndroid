registerReceiver

@JvmOverloads
fun Context.registerReceiver(filter: IntentFilter, flags: Int? = null, exported: Boolean = true, body: BroadcastReceiver.(context: Context, intent: Intent) -> Unit): BroadcastReceiver

Register a broadcast receiver.

Usage:

val filter = IntentFilter().apply {
    addAction("com.example.app.action.KNOCK")
}
val receiver = registerReceiver(filter) { context, intent ->
    val greetings = intent.getStringExtra("greetings")
    context.toast(greetings)
}
// Unregister the receiver.
unregisterReceiver(receiver)
