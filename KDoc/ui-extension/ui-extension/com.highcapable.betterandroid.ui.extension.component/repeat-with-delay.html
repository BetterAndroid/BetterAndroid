fun CoroutineScope.repeatWithDelay(times: Int, downTo: Int = 0, timeMillis: Long = 1000, context: CoroutineContext = EmptyCoroutineContext, withContext: CoroutineDispatcher = Dispatchers.Main, block: (index: Int) -> Unit): Job

Start a coroutine task with a countdown.

Usage:

// Countdown from 10 to 0, delay 1000 milliseconds each time.
repeatWithDelay(10) { index ->
    Log.d("Repeat", "Index: $index")
}
