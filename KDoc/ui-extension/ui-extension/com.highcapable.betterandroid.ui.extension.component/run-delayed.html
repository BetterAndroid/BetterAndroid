fun CoroutineScope.runDelayed(timeMillis: Long = 150, context: CoroutineContext = EmptyCoroutineContext, withContext: CoroutineDispatcher = Dispatchers.Main, block: () -> Unit): Job

Start delayed coroutine task.

Usage:

// Delay 500 milliseconds to run the block.
// For the original android Handler usage.
Handler().postDelayed({
    // Do something.
}, 500)
// For this function usage.
runDelayed(500) {
    // Do something.
}
