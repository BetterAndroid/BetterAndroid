viewBinding

inline fun <VB : ViewBinding> Context.viewBinding(parent: ViewGroup? = null, attachToParent: Boolean = false): ViewBindingDelegate<VB>

Create a delegate for ViewBinding.

Usage:

class YourActivity : Activity() {

    val binding: ActivityMainBinding by viewBinding()

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(binding.root)
    }
}

    val binding: FragmentMainBinding by viewBinding()

    override fun onCreateView(
        inflater: LayoutInflater,
        container: ViewGroup?,
        savedInstanceState: Bundle?
